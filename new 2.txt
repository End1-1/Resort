select od.store_id, md.name, sum(od.qty), sum(od.qty*od.price)
from o_dishes od, o_order oo, me_dishes md
where od.state_id=1 and oo.state_id=2 and oo.date_cash between '01.01.2017' and '01.01.2018' and od.order_id=oo.id
group by 1, 2
order by 3 desc
